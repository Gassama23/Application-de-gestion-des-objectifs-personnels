
ALGORITHME genererPlanningAdapte(utilisateur, objectif)

DÉBUT

    planning ← NOUVEAU Planning()
    attentes ← NOUVEAU AttentesUtilisateur()

    SELON objectif.getCategorie()

        CAS ECONOMIE :

            AFFICHER "Quel est votre revenu mensuel actuel ?"
            attentes.revenuMensuel ← LIRE_DOUBLE()

            AFFICHER "Combien épargnez-vous actuellement par mois ?"
            attentes.epargneActuelle ← LIRE_DOUBLE()

            AFFICHER "Quel est votre objectif d'épargne ?"
            attentes.montantCible ← LIRE_DOUBLE()

            AFFICHER "En combien de mois ?"
            attentes.delaiMois ← LIRE_ENTIER()

            AFFICHER "Quel est votre comportement financier ?"
            attentes.comportementFinancier ← LIRE_ENTIER()

            AFFICHER "Quelles sont vos principales dépenses ?"
            attentes.categoriesDepenses ← LIRE_LISTE_CHAINES()

            AFFICHER "Avez-vous des dettes ?"
            attentes.situationDettes ← LIRE_ENTIER()

            SI attentes.situationDettes <= 2 ALORS
                AFFICHER "Montant total des dettes"
                attentes.montantDettes ← LIRE_DOUBLE()
            FIN SI

            AFFICHER "Avez-vous un fonds d'urgence ?"
            attentes.fondsUrgence ← LIRE_ENTIER()

            epargneJournaliere ← attentes.montantCible / (attentes.delaiMois * 30)

            epargneRecommandee ← epargneJournaliere * 1.1

            tauxEpargne ← (attentes.epargneActuelle / attentes.revenuMensuel) * 100

            tauxCible ← ((epargneRecommandee * 30) / attentes.revenuMensuel) * 100

            planning.titre ← "Plan d'épargne personnalisé"

            planning.AJOUTER_SECTION("DIAGNOSTIC")

            planning.AJOUTER_LIGNE("Revenu : " + attentes.revenuMensuel)

            planning.AJOUTER_LIGNE("Épargne actuelle : " + attentes.epargneActuelle)

            planning.AJOUTER_LIGNE("Objectif : " + attentes.montantCible)

            planning.AJOUTER_LIGNE("Délai : " + attentes.delaiMois)

            planning.AJOUTER_LIGNE("Épargne journalière : " + ARRONDIR(epargneJournaliere))

            planning.AJOUTER_LIGNE("Taux cible : " + ARRONDIR(tauxCible))

            planning.AJOUTER_SECTION("RECOMMANDATIONS")

            SI attentes.fondsUrgence == 3 ALORS

                fondsUrgenceMin ← attentes.revenuMensuel * 3

                planning.AJOUTER_LIGNE("Créer un fonds d'urgence")

                planning.AJOUTER_LIGNE("Montant minimum : " + fondsUrgenceMin)

            FIN SI

            SI attentes.situationDettes == 1 ALORS

                planning.AJOUTER_LIGNE("Réduire les dettes en priorité")

            FIN SI

            SI attentes.comportementFinancier == 1 ALORS

                planning.AJOUTER_LIGNE("Tenir un journal des dépenses")

            FIN SI

            planning.AJOUTER_SECTION("PLAN D'ACTION")

            planning.AJOUTER_LIGNE("Épargner chaque jour")

            planning.AJOUTER_LIGNE("Noter les dépenses")

            planning.AJOUTER_LIGNE("Réduire les dépenses inutiles")

            SI tauxCible > 30 ALORS

                planning.AJOUTER_LIGNE("Chercher des revenus supplémentaires")

            FIN SI

        CAS SPORT :

            AFFICHER "Quel est votre objectif sportif ?"
            attentes.objectifSport ← LIRE_ENTIER()

            SI attentes.objectifSport == 1 ALORS

                AFFICHER "Combien de kilos souhaitez-vous perdre ?"
                attentes.poidsAPerdre ← LIRE_DOUBLE()

            SINON SI attentes.objectifSport == 2 ALORS

                AFFICHER "Combien de kilos de muscle souhaitez-vous gagner ?"
                attentes.muscleCible ← LIRE_DOUBLE()

            FIN SI

            AFFICHER "Quel est votre niveau ?"
            attentes.niveau ← LIRE_ENTIER()

            AFFICHER "Combien de séances par semaine ?"
            attentes.frequenceHebdo ← LIRE_ENTIER()

            AFFICHER "Durée des séances ?"
            attentes.dureeSeance ← LIRE_ENTIER()

            AFFICHER "Accès à une salle de sport ?"
            attentes.accesSalle ← (LIRE_ENTIER() == 1)

            AFFICHER "Type d'entraînement préféré ?"
            attentes.typeEntrainement ← LIRE_ENTIER()

            AFFICHER "Contraintes de santé ?"
            attentes.contraintesSante ← LIRE_ENTIER()

            planning.titre ← "Programme sportif personnalisé"

            SI attentes.objectifSport == 1 ALORS

                semainesNecessaires ← attentes.poidsAPerdre / 0.5

                caloriesParSeance ← 400

            FIN SI

            SI attentes.niveau == 1 ALORS

                intensite ← "FAIBLE"

            SINON SI attentes.niveau == 2 ALORS

                intensite ← "MODÉRÉE"

            SINON

                intensite ← "ÉLEVÉE"

            FIN SI

            planning.AJOUTER_SECTION("PROFIL")

            planning.AJOUTER_LIGNE("Objectif : " + TRADUIRE_OBJECTIF(attentes.objectifSport))

            planning.AJOUTER_LIGNE("Niveau : " + TRADUIRE_NIVEAU(attentes.niveau))

            planning.AJOUTER_LIGNE("Fréquence : " + attentes.frequenceHebdo)

            planning.AJOUTER_LIGNE("Durée : " + attentes.dureeSeance)

            planning.AJOUTER_LIGNE("Intensité : " + intensite)

            planning.AJOUTER_SECTION("PLANNING HEBDOMADAIRE")

            joursEntrainement ← REPARTIR_JOURS(attentes.frequenceHebdo)

            POUR i DE 1 À 7 FAIRE

                SI i DANS joursEntrainement ALORS

                    seance ← GENERER_SEANCE(attentes, i, intensite)

                    planning.AJOUTER_LIGNE("Jour " + i + " : " + seance.nom)

                SINON

                    planning.AJOUTER_LIGNE("Jour " + i + " : Repos")

                FIN SI

            FIN POUR

        CAS APPRENTISSAGE :

            AFFICHER "Que souhaitez-vous apprendre ?"
            attentes.sujet ← LIRE_CHAINE()

            AFFICHER "Quel est votre niveau actuel ?"
            attentes.niveau ← LIRE_ENTIER()

            AFFICHER "Quel est votre objectif final ?"
            attentes.objectifFinal ← LIRE_ENTIER()

            AFFICHER "Temps disponible par jour ?"
            attentes.tempsQuotidien ← LIRE_ENTIER()

            AFFICHER "Style d'apprentissage préféré ?"
            attentes.styleApprentissage ← LIRE_ENTIER()

            AFFICHER "Avez-vous déjà abandonné un apprentissage similaire ?"
            attentes.historique ← LIRE_ENTIER()

            planning.AJOUTER_SECTION("PARCOURS D'APPRENTISSAGE")

            SI attentes.historique == 1 ALORS

                planning.AJOUTER_LIGNE("Commencer progressivement")

            FIN SI

            planning.AJOUTER_SECTION("PROGRAMME")

            planning.AJOUTER_LIGNE("Étudier chaque jour")

            planning.AJOUTER_LIGNE("Réviser régulièrement")

            planning.AJOUTER_LIGNE("Faire des exercices pratiques")

        CAS DEVELOPPEMENT_PERSONNEL :

            planning.AJOUTER_SECTION("DÉVELOPPEMENT PERSONNEL")

            planning.AJOUTER_LIGNE("Mettre en place des habitudes")

            planning.AJOUTER_LIGNE("Créer une routine quotidienne")

            planning.AJOUTER_LIGNE("Définir des micro-objectifs")

    FIN SELON

    actionsQuotidiennes ← EXTRAIRE_ACTIONS(planning, attentes)

    POUR CHAQUE action DANS actionsQuotidiennes FAIRE

        actionQuotidienneRepository.SAUVEGARDER(action)

    FIN POUR

    rappels ← GENERER_RAPPELS(attentes, objectif)

    POUR CHAQUE rappel DANS rappels FAIRE

        rappelRepository.SAUVEGARDER(rappel)

    FIN POUR

    planning.objectifId ← objectif.getId()

    planning.utilisateurId ← utilisateur.getId()

    planning.dateCreation ← AUJOURD_HUI()

    planning.attentes ← attentes

    planningRepository.SAUVEGARDER(planning)

    AFFICHER planning.FORMATER_POUR_AFFICHAGE()

    RETOURNER planning

FIN
