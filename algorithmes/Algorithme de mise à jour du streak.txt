VARIABLES
  aRealiseActionAujourdHui EST_DU_TYPE NOMBRE
  streakActuel EST_DU_TYPE NOMBRE
  meilleurStreak EST_DU_TYPE NOMBRE

DEBUT_ALGORITHME

  // Simulation des données actuelles de l'utilisateur (Exemples)
  // streakActuel POUR_PRENDRE_LA_VALEUR 5
  // meilleurStreak POUR_PRENDRE_LA_VALEUR 12

  AFFICHER "L'utilisateur a-t-il réalisé une action aujourd'hui ? (1:VRAI, 0:FAUX)"
  LIRE aRealiseActionAujourdHui

  SI (aRealiseActionAujourdHui == 1) ALORS
    DEBUT_SI
    streakActuel POUR_PRENDRE_LA_VALEUR streakActuel + 1
    
    // Vérification du record de streak
    SI (streakActuel > meilleurStreak) ALORS
      DEBUT_SI
      meilleurStreak POUR_PRENDRE_LA_VALEUR streakActuel
      FIN_SI
    FIN_SI
  SINON
    DEBUT_SINON
    streakActuel POUR_PRENDRE_LA_VALEUR 0
    FIN_SINON

  // Affichage du résultat (Simulation de utilisateurRepository.METTRE_A_JOUR)
  AFFICHER "=== Mise à jour de l'utilisateur ==="
  AFFICHER "Streak actuel : "
  AFFICHER* streakActuel
  AFFICHER "Meilleur streak : "
  AFFICHER* meilleurStreak

FIN_ALGORITHME
