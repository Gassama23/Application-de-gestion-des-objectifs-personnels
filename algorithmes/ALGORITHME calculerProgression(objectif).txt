VARIABLES
  actionsTotal EST_DU_TYPE NOMBRE
  actionsTerminees EST_DU_TYPE NOMBRE
  progression EST_DU_TYPE NOMBRE

DEBUT_ALGORITHME
  // Simulation des données de l'objectif (Exemples de valeurs)
  // actionsTotal POUR_PRENDRE_LA_VALEUR 10
  // actionsTerminees POUR_PRENDRE_LA_VALEUR 4

  SI (actionsTotal <= 0) ALORS
    DEBUT_SI
    progression POUR_PRENDRE_LA_VALEUR 0
    FIN_SI
  SINON
    DEBUT_SINON
    progression POUR_PRENDRE_LA_VALEUR (actionsTerminees * 100) / actionsTotal
    FIN_SINON

  // Sauvegarde du résultat (Simulation de objectif.progression)
  AFFICHER "Progression de l'objectif : "
  AFFICHER progression
  AFFICHER " %"

FIN_ALGORITHME
