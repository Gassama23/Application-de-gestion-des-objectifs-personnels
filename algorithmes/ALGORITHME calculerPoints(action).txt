VARIABLES
  points EST_DU_TYPE NOMBRE
  estRealisee EST_DU_TYPE NOMBRE
  estEffectueeAvantHeure EST_DU_TYPE NOMBRE

DEBUT_ALGORITHME

  // Initialisation des points
  points POUR_PRENDRE_LA_VALEUR 0

  // Demande des états de l'action pour simulation
  AFFICHER "L'action est-elle réalisée ? (1 pour Oui, 0 pour Non)"
  LIRE estRealisee
  AFFICHER "L'action est-elle effectuée avant l'heure ? (1 pour Oui, 0 pour Non)"
  LIRE estEffectueeAvantHeure

  // Condition 1 : Action réalisée
  SI (estRealisee == 1) ALORS
    DEBUT_SI
    points POUR_PRENDRE_LA_VALEUR points + 10
    FIN_SI

  // Condition 2 : Action effectuée en avance
  SI (estEffectueeAvantHeure == 1) ALORS
    DEBUT_SI
    points POUR_PRENDRE_LA_VALEUR points + 5
    FIN_SI

  // Affichage du résultat final
  AFFICHER "Points calculés : "
  AFFICHER points

FIN_ALGORITHME
