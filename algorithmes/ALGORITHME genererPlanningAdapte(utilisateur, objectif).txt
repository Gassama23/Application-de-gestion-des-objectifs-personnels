VARIABLES
  // Variables de contrôle et d'embranchement
  categorieObjectif EST_DU_TYPE NOMBRE
  i EST_DU_TYPE NOMBRE
  
  // Variables pour le CAS 1 : ÉCONOMIE
  revenuMensuel EST_DU_TYPE NOMBRE
  epargneActuelle EST_DU_TYPE NOMBRE
  montantCible EST_DU_TYPE NOMBRE
  delaiMois EST_DU_TYPE NOMBRE
  comportementFinancier EST_DU_TYPE NOMBRE
  situationDettes EST_DU_TYPE NOMBRE
  montantDettes EST_DU_TYPE NOMBRE
  fondsUrgence EST_DU_TYPE NOMBRE
  epargneJournaliere EST_DU_TYPE NOMBRE
  epargneRecommandee EST_DU_TYPE NOMBRE
  tauxEpargne EST_DU_TYPE NOMBRE
  tauxCible EST_DU_TYPE NOMBRE
  fondsUrgenceMin EST_DU_TYPE NOMBRE
  
  // Variables pour le CAS 2 : SPORT
  objectifSport EST_DU_TYPE NOMBRE
  poidsAPerdre EST_DU_TYPE NOMBRE
  muscleCible EST_DU_TYPE NOMBRE
  niveau EST_DU_TYPE NOMBRE
  frequenceHebdo EST_DU_TYPE NOMBRE
  dureeSeance EST_DU_TYPE NOMBRE
  accesSalle EST_DU_TYPE NOMBRE
  typeEntrainement EST_DU_TYPE NOMBRE
  contraintesSante EST_DU_TYPE NOMBRE
  semainesNecessaires EST_DU_TYPE NOMBRE
  caloriesParSeance EST_DU_TYPE NOMBRE
  
  // Variables pour le CAS 3 : APPRENTISSAGE
  sujet EST_DU_TYPE CHAINE
  objectifFinal EST_DU_TYPE NOMBRE
  tempsQuotidien EST_DU_TYPE NOMBRE
  styleApprentissage EST_DU_TYPE NOMBRE
  historique EST_DU_TYPE NOMBRE

DEBUT_ALGORITHME

  // Sélection manuelle de la catégorie pour simulation AlgoBox
  AFFICHER "Sélectionnez la catégorie (1:ÉCONOMIE, 2:SPORT, 3:APPRENTISSAGE, 4:DEV_PERSO) :"
  LIRE categorieObjectif

  // --- CAS 1 : ÉCONOMIE ---
  SI (categorieObjectif == 1) ALORS
    DEBUT_SI
    AFFICHER "Quel est votre revenu mensuel actuel ?"
    LIRE revenuMensuel
    AFFICHER "Combien épargnez-vous actuellement par mois ?"
    LIRE epargneActuelle
    AFFICHER "Quel est votre objectif d'épargne ?"
    LIRE montantCible
    AFFICHER "En combien de mois ?"
    LIRE delaiMois
    AFFICHER "Quel est votre comportement financier (1:Impulsif, 2:Modéré, 3:Économe) ?"
    LIRE comportementFinancier
    AFFICHER "Note : (Saisie de liste ignorée sur AlgoBox)"
    AFFICHER "Avez-vous des dettes (1:Oui prioritaires, 2:Oui secondaires, 3:Non) ?"
    LIRE situationDettes
    
    SI (situationDettes <= 2) ALORS
      DEBUT_SI
      AFFICHER "Montant total des dettes ?"
      LIRE montantDettes
      FIN_SI
      
    AFFICHER "Avez-vous un fonds d'urgence (1:Oui, 2:En cours, 3:Non) ?"
    LIRE fondsUrgence
    
    // Calculs financiers
    epargneJournaliere POUR_PRENDRE_LA_VALEUR montantCible / (delaiMois * 30)
    epargneRecommandee POUR_PRENDRE_LA_VALEUR epargneJournaliere * 1.1
    tauxEpargne POUR_PRENDRE_LA_VALEUR (epargneActuelle / revenuMensuel) * 100
    tauxCible POUR_PRENDRE_LA_VALEUR ((epargneRecommandee * 30) / revenuMensuel) * 100
    
    // Affichage du Diagnostic
    AFFICHER "=== Plan d'épargne personnalisé ==="
    AFFICHER "--- DIAGNOSTIC ---"
    AFFICHER "Revenu : "
    AFFICHER* revenuMensuel
    AFFICHER "Épargne actuelle : "
    AFFICHER* epargneActuelle
    AFFICHER "Objectif : "
    AFFICHER* montantCible
    AFFICHER "Délai : "
    AFFICHER* delaiMois
    AFFICHER "Épargne journalière : "
    AFFICHER* round(epargneJournaliere)
    AFFICHER "Taux cible : "
    AFFICHER* round(tauxCible)
    
    // Affichage des Recommandations
    AFFICHER "--- RECOMMANDATIONS ---"
    SI (fondsUrgence == 3) ALORS
      DEBUT_SI
      fondsUrgenceMin POUR_PRENDRE_LA_VALEUR revenuMensuel * 3
      AFFICHER "-> Créer un fonds d'urgence"
      AFFICHER "   Montant minimum : "
      AFFICHER* fondsUrgenceMin
      FIN_SI
    SI (situationDettes == 1) ALORS
      DEBUT_SI
      AFFICHER "-> Réduire les dettes en priorité"
      FIN_SI
    SI (comportementFinancier == 1) ALORS
      DEBUT_SI
      AFFICHER "-> Tenir un journal des dépenses"
      FIN_SI
      
    // Plan d'action
    AFFICHER "--- PLAN D'ACTION ---"
    AFFICHER "-> Épargner chaque jour"
    AFFICHER "-> Noter les dépenses"
    AFFICHER "-> Réduire les dépenses inutiles"
    SI (tauxCible > 30) ALORS
      DEBUT_SI
      AFFICHER "-> Chercher des revenus supplémentaires"
      FIN_SI
    FIN_SI
  SINON
    DEBUT_SINON
    
    // --- CAS 2 : SPORT ---
    SI (categorieObjectif == 2) ALORS
      DEBUT_SI
      AFFICHER "Quel est votre objectif sportif (1:Perte de poids, 2:Prise de muscle) ?"
      LIRE objectifSport
      
      SI (objectifSport == 1) ALORS
        DEBUT_SI
        AFFICHER "Combien de kilos souhaitez-vous perdre ?"
        LIRE poidsAPerdre
        semainesNecessaires POUR_PRENDRE_LA_VALEUR poidsAPerdre / 0.5
        caloriesParSeance POUR_PRENDRE_LA_VALEUR 400
        FIN_SI
      SINON
        DEBUT_SINON
        SI (objectifSport == 2) ALORS
          DEBUT_SI
          AFFICHER "Combien de kilos de muscle souhaitez-vous gagner ?"
          LIRE muscleCible
          FIN_SI
        FIN_SINON
        
      AFFICHER "Quel est votre niveau (1:Débutant, 2:Intermédiaire, 3:Avancé) ?"
      LIRE niveau
      AFFICHER "Combien de séances par semaine ?"
      LIRE frequenceHebdo
      AFFICHER "Durée des séances (en minutes) ?"
      LIRE dureeSeance
      AFFICHER "Accès à une salle de sport (1:Oui, 2:Non) ?"
      LIRE accesSalle
      AFFICHER "Type d'entraînement préféré ?"
      LIRE typeEntrainement
      AFFICHER "Contraintes de santé ?"
      LIRE contraintesSante
      
      AFFICHER "=== Programme sportif personnalisé ==="
      AFFICHER "--- PROFIL ---"
      AFFICHER "Objectif : "
      AFFICHER* objectifSport
      AFFICHER "Niveau : "
      AFFICHER* niveau
      AFFICHER "Fréquence : "
      AFFICHER* frequenceHebdo
      AFFICHER "Durée : "
      AFFICHER* dureeSeance
      
      SI (niveau == 1) ALORS
        DEBUT_SI
        AFFICHER "Intensité : FAIBLE"
        FIN_SI
      SINON
        DEBUT_SINON
        SI (niveau == 2) ALORS
          DEBUT_SI
          AFFICHER "Intensité : MODÉRÉE"
          FIN_SI
        SINON
          DEBUT_SINON
          AFFICHER "Intensité : ÉLEVÉE"
          FIN_SINON
        FIN_SINON
        
      // Génération simplifiée du planning hebdomadaire
      AFFICHER "--- PLANNING HEBDOMADAIRE ---"
      AFFICHER "Note : (Répartition automatisée non supportée nativement)"
      POUR i POUR_PRENDRE_LA_VALEUR 1 ALLANT_A 7
        DEBUT_POUR
        AFFICHER "Jour "
        AFFICHER i
        AFFICHER " : Voir détails dans l'application"
        FIN_POUR
      FIN_SI
    SINON
      DEBUT_SINON
      
      // --- CAS 3 : APPRENTISSAGE ---
      SI (categorieObjectif == 3) ALORS
        DEBUT_SI
        AFFICHER "Que souhaitez-vous apprendre ?"
        LIRE sujet
        AFFICHER "Quel est votre niveau actuel (1 à 5) ?"
        LIRE niveau
        AFFICHER "Quel est votre objectif final ?"
        LIRE objectifFinal
        AFFICHER "Temps disponible par jour (minutes) ?"
        LIRE tempsQuotidien
        AFFICHER "Style d'apprentissage préféré ?"
        LIRE styleApprentissage
        AFFICHER "Avez-vous déjà abandonné un apprentissage similaire (1:Oui, 2:Non) ?"
        LIRE historique
        
        AFFICHER "=== PARCOURS D'APPRENTISSAGE ==="
        SI (historique == 1) ALORS
          DEBUT_SI
          AFFICHER "-> Commencer progressivement"
          FIN_SI
          
        AFFICHER "--- PROGRAMME ---"
        AFFICHER "-> Étudier chaque jour"
        AFFICHER "-> Réviser régulièrement"
        AFFICHER "-> Faire des exercices pratiques"
        FIN_SI
      SINON
        DEBUT_SINON
        
        // --- CAS 4 : DÉVELOPPEMENT PERSONNEL ---
        SI (categorieObjectif == 4) ALORS
          DEBUT_SI
          AFFICHER "=== DÉVELOPPEMENT PERSONNEL ==="
          AFFICHER "-> Mettre en place des habitudes"
          AFFICHER "-> Créer une routine quotidienne"
          AFFICHER "-> Définir des micro-objectifs"
          FIN_SI
        FIN_SINON
        
      FIN_SINON
    FIN_SINON
  FIN_SINON

  // Traitements persistants (Simulés en fin d'exécution)
  AFFICHER "Données de planification et rappels synchronisés en base de données."

FIN_ALGORITHME
